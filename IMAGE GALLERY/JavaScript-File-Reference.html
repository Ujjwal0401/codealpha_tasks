/*
============================================================
PROFESSIONAL IMAGE GALLERY - JAVASCRIPT
============================================================
Features:
- Category filtering system
- Lightbox modal with image navigation
- Keyboard controls (arrow keys, escape)
- Touch/swipe support for mobile devices
- Smooth animations and transitions
- Responsive behavior
============================================================
*/

// ==================== DOM ELEMENTS ==================== //
const galleryImages = document.querySelectorAll('.gallery-item');
const filterButtons = document.querySelectorAll('.filter-btn');
const lightbox = document.getElementById('lightbox');
const lightboxImage = document.querySelector('.lightbox-image');
const lightboxCaption = document.querySelector('.lightbox-caption');
const lightboxClose = document.querySelector('.lightbox-close');
const lightboxPrev = document.querySelector('.lightbox-prev');
const lightboxNext = document.querySelector('.lightbox-next');

// ==================== VARIABLES ==================== //
let currentImageIndex = 0;      // Track current image index
let filteredImages = [];        // Store filtered images
let touchStartX = 0;            // Touch start position
let touchEndX = 0;              // Touch end position

// ==================== FILTER FUNCTIONALITY ==================== //

/**
 * Initialize filter button event listeners
 * Adds click handlers to all filter buttons
 */
filterButtons.forEach(button => {
    button.addEventListener('click', () => {
        // Remove active class from all buttons
        filterButtons.forEach(btn => btn.classList.remove('active'));
        
        // Add active class to clicked button
        button.classList.add('active');
        
        // Get filter value from data attribute
        const filterValue = button.getAttribute('data-filter');
        
        // Call filter function
        filterImages(filterValue);
    });
});

/**
 * Filter and display images based on category
 * @param {string} category - The category to filter by ('all', 'nature', 'architecture', 'people')
 */
function filterImages(category) {
    galleryImages.forEach(item => {
        if (category === 'all') {
            // Show all images
            item.classList.remove('hidden');
            item.style.display = 'block';
            item.style.animation = 'fadeIn 0.5s ease forwards';
        } else {
            // Show only images with matching category
            if (item.getAttribute('data-category') === category) {
                item.classList.remove('hidden');
                item.style.display = 'block';
                item.style.animation = 'fadeIn 0.5s ease forwards';
            } else {
                item.classList.add('hidden');
                item.style.display = 'none';
            }
        }
    });
}

// ==================== LIGHTBOX FUNCTIONALITY ==================== //

/**
 * Initialize gallery item click listeners
 * Opens lightbox when an image is clicked
 */
galleryImages.forEach((item, index) => {
    item.addEventListener('click', () => {
        // Check if item is visible (not filtered out)
        if (item.style.display !== 'none') {
            openLightbox(item);
        }
    });
});

/**
 * Open the lightbox with the clicked image
 * @param {HTMLElement} item - The gallery item to display
 */
function openLightbox(item) {
    const img = item.querySelector('img');
    const caption = item.querySelector('h3');
    
    // Set lightbox content
    lightboxImage.src = img.src;
    lightboxCaption.textContent = caption.textContent;
    
    // Show lightbox
    lightbox.classList.add('active');
    document.body.style.overflow = 'hidden';
    
    // Store current index
    currentImageIndex = Array.from(galleryImages).indexOf(item);
}

/**
 * Close the lightbox modal
 */
function closeLightbox() {
    lightbox.classList.remove('active');
    lightboxImage.src = '';
    document.body.style.overflow = 'auto';
}

// ==================== EVENT LISTENERS ==================== //

// Close button click event
lightboxClose.addEventListener('click', closeLightbox);

// Click outside the image to close lightbox
lightbox.addEventListener('click', (e) => {
    if (e.target === lightbox) {
        closeLightbox();
    }
});

/**
 * Keyboard navigation for lightbox
 * - Escape: Close lightbox
 * - Arrow Left: Previous image
 * - Arrow Right: Next image
 */
document.addEventListener('keydown', (e) => {
    if (!lightbox.classList.contains('active')) return;
    
    if (e.key === 'Escape') {
        closeLightbox();
    } else if (e.key === 'ArrowLeft') {
        showPreviousImage();
    } else if (e.key === 'ArrowRight') {
        showNextImage();
    }
});

// Navigation button events
lightboxPrev.addEventListener('click', showPreviousImage);
lightboxNext.addEventListener('click', showNextImage);

// ==================== IMAGE NAVIGATION ==================== //

/**
 * Display the previous image in the lightbox
 * Considers filtered images only
 */
function showPreviousImage() {
    // Get all visible (filtered) images
    const visibleItems = Array.from(galleryImages).filter(item => 
        item.style.display !== 'none' && !item.classList.contains('hidden')
    );
    
    // Find current image index in visible items
    let currentVisibleIndex = visibleItems.findIndex(item => {
        const img = item.querySelector('img');
        return img.src === lightboxImage.src;
    });
    
    // Calculate previous index (loop around)
    currentVisibleIndex = (currentVisibleIndex - 1 + visibleItems.length) % visibleItems.length;
    
    // Update lightbox with previous image
    const prevItem = visibleItems[currentVisibleIndex];
    updateLightboxImage(prevItem);
}

/**
 * Display the next image in the lightbox
 * Considers filtered images only
 */
function showNextImage() {
    // Get all visible (filtered) images
    const visibleItems = Array.from(galleryImages).filter(item => 
        item.style.display !== 'none' && !item.classList.contains('hidden')
    );
    
    // Find current image index in visible items
    let currentVisibleIndex = visibleItems.findIndex(item => {
        const img = item.querySelector('img');
        return img.src === lightboxImage.src;
    });
    
    // Calculate next index (loop around)
    currentVisibleIndex = (currentVisibleIndex + 1) % visibleItems.length;
    
    // Update lightbox with next image
    const nextItem = visibleItems[currentVisibleIndex];
    updateLightboxImage(nextItem);
}

/**
 * Update the lightbox image and caption
 * @param {HTMLElement} item - The gallery item to display
 */
function updateLightboxImage(item) {
    const img = item.querySelector('img');
    const caption = item.querySelector('h3');
    
    // Reset animation to trigger it again
    lightboxImage.style.animation = 'none';
    
    // Update after brief delay for animation reset
    setTimeout(() => {
        lightboxImage.src = img.src;
        lightboxCaption.textContent = caption.textContent;
        lightboxImage.style.animation = 'zoomIn 0.4s ease';
    }, 10);
}

// ==================== TOUCH/SWIPE SUPPORT ==================== //

/**
 * Handle touch start - record initial X position
 */
lightbox.addEventListener('touchstart', (e) => {
    touchStartX = e.changedTouches[0].screenX;
});

/**
 * Handle touch end - calculate swipe distance
 */
lightbox.addEventListener('touchend', (e) => {
    touchEndX = e.changedTouches[0].screenX;
    handleSwipe();
});

/**
 * Process swipe gesture
 * Left swipe: Next image
 * Right swipe: Previous image
 */
function handleSwipe() {
    const swipeThreshold = 50;  // Minimum pixels to register as swipe
    
    if (touchStartX - touchEndX > swipeThreshold) {
        // Swiped left - show next image
        showNextImage();
    } else if (touchEndX - touchStartX > swipeThreshold) {
        // Swiped right - show previous image
        showPreviousImage();
    }
}

window.addEventListener('load', () => {
    console.log('%c✨ Premium Image Gallery Loaded Successfully! ✨', 
        'color: #667eea; font-size: 16px; font-weight: bold;');
    console.log(`Total Images: ${galleryImages.length}`);
    console.log('Features:');
    console.log('  ✓ Category Filtering');
    console.log('  ✓ Lightbox View');
    console.log('  ✓ Keyboard Navigation (← → Escape)');
    console.log('  ✓ Touch/Swipe Support');
    console.log('  ✓ Responsive Design');
});
